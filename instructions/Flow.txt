# 1) Install
cd server && npm i
cd ../client && npm i
cd ..

# 2) Env files
cp .env.example client/.env.local
cp server/.env.example server/.env
# Fill in values as described earlier (Firebase config + admin emails)

# 3) Start backend
cd server
npm run dev
# Server on http://localhost:5000

# 4) Start frontend (new terminal)
cd client
npm run dev
# Vite on http://localhost:5173
